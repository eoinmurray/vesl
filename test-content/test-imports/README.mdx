# Testing npm Package Imports

This tests importing from installed npm packages, not just local files.

## React useState

export const Counter = () => {
  const [count, setCount] = useState(0)
  return (
    <button
      onClick={() => setCount(c => c + 1)}
      className="px-4 py-2 bg-primary text-primary-foreground rounded-md"
    >
      Count: {count}
    </button>
  )
}
